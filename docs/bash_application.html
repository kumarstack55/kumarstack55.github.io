ShellCheck をインストールする。¶

cd $HOME
curl -Lo - https://github.com/koalaman/shellcheck/releases/download/stable/shellcheck-stable.linux.x86_64.tar.xz | tar -Jxvf -
ln -s $HOME/shellcheck-stable/shellcheck ~/bin/shellcheck

異常終了する。¶

#!/bin/bash

# shellcheck source=err.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/err.sh"

die() {
  local msg="${1:-Died}"

  local at=""
  local stack
  # shellcheck disable=SC2207
  if stack=($(caller 0)); then
    local line="${stack[0]}" file="${stack[2]}"
    at=" at ${file} line ${line}"
  fi
  err "${msg}${at}."
  exit 1
}

#!/bin/bash

# shellcheck source=lib/die.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/die.sh"

f() {
  false
}

f || die "Failed to execute f()"

$ ./demo_die.sh
Failed to execute f() at ./demo_die.sh line 10.

ログを出力する。¶

#!/bin/bash

# shellcheck source=err.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/err.sh"

_LOG_STDERR=1
_LOG_SYSLOG=1
_LOG_SYSLOG_FACILITY=local0
_LOG_NAME=$(basename "$0")

log_disable_stderr() {
  _LOG_STDERR=''
}

log_disable_syslog() {
  _LOG_SYSLOG=''
}

log_disable_syslog_facility() {
  _LOG_SYSLOG_FACILITY="$1"
}

_log_pri() {
  echo "${_LOG_SYSLOG_FACILITY}.${2:-err}"
}

_log() {
  [ "$_LOG_STDERR" ] && err "$(date --rfc-3339=seconds) $_LOG_NAME $2"
  if [ "$_LOG_SYSLOG" ]; then
    local pri
    pri=$(_log_pri "$1") || die
    logger -p "$pri" -t "$_LOG_NAME" "$2"
  fi
}

log() {
  _log '' "$1"
}

log_crit() {
  _log 'crit' "$1"
}

log_err() {
  _log 'err' "$1"
}

log_warn() {
  _log 'warn' "$1"
}

log_info() {
  _log 'info' "$1"
}

log_debug() {
  _log 'debug' "$1"
}

#!/bin/bash

# shellcheck source=lib/log.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/log.sh"

log_err "test"

$ ./demo_log.sh
2021-12-07 00:40:24+09:00 demo_log.sh test

スタックトレースを出力する。¶

#!/bin/bash

# shellcheck source=err.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/err.sh"

print_stacktrace() {
  local index=0
  local stack
  err "Traceback (most recent call first):"
  # shellcheck disable=SC2207
  while stack=($(caller "${index}")); do
    local line="${stack[0]}" sub="${stack[1]}" file="${stack[2]}"
    err "  ${file}, line ${line}, in ${sub}"
    ((index++))
  done
}

#!/bin/bash

# shellcheck source=lib/die.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/print_stacktrace.sh"

f() {
  print_stacktrace
}

g() {
  f
}

g

$ ./demo_print_stacktrace.sh
Traceback (most recent call first):
  ./demo_print_stacktrace.sh, line 7, in f
  ./demo_print_stacktrace.sh, line 11, in g
  ./demo_print_stacktrace.sh, line 14, in main

#!/bin/bash

# shellcheck source=die.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/die.sh"

ensure() {
  if ! "$@"; then die "command failed: $*"; fi
}

#!/bin/bash

# shellcheck source=lib/ensure.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/ensure.sh"

f() {
  false
}

左端の空白を除去する。¶

f() {
  local msg="$1"; shift
  echo "${msg#"${msg%%[![:space:]]*}"}"
}

f "a b c"
