cd $HOME
curl -Lo - https://github.com/koalaman/shellcheck/releases/download/stable/shellcheck-stable.linux.x86_64.tar.xz | tar -Jxvf -
ln -s $HOME/shellcheck-stable/shellcheck ~/bin/shellcheck

ログを出力する。¶

snippets/bash/lib/log.sh¶
#!/bin/bash

# shellcheck source=err.sh
. "${HIY_LIB:-./lib}/err.sh"

_log_stderr=1
_log_syslog=1
_log_syslog_facility=local0
_log_name=$(basename "$0")

log_disable_stderr() {
  _log_stderr=''
}

log_disable_syslog() {
  _log_syslog=''
}

log_set_syslog_facility() {
  _log_syslog_facility="$1"
}

_log_pri() {
  echo "${_log_syslog_facility}.${2:-err}"
}

_log() {
  local sec="${3:-$(date +%s)}"
  [ "$_log_stderr" ] \
    && err "$(date --rfc-3339=seconds --date="@$sec") $_log_name $2"
  if [ "$_log_syslog" ]; then
    local pri
    pri=$(_log_pri "$1")
    logger -p "$pri" -t "$_log_name" "$2"
  fi
}

log() {
  _log '' "$1" "$2"
}

log_crit() {
  _log 'crit' "$1" "$2"
}

log_err() {
  _log 'err' "$1" "$2"
}

log_warn() {
  _log 'warn' "$1" "$2"
}

log_info() {
  _log 'info' "$1" "$2"
}

log_debug() {
  _log 'debug' "$1" "$2"
}

異常終了する。¶

snippets/bash/lib/die.sh¶
#!/bin/bash

# shellcheck source=err.sh
. "${HIY_LIB:-./lib}/err.sh"

die() {
  local msg="${1:-Died}"

  local stack
  # shellcheck disable=SC2207
  if stack=($(caller 0)); then
    err "${msg} at ${stack[2]} line ${stack[0]}."
  else
    err "${msg}."
  fi
  exit 1
}

チェックモードで実行する。¶

snippets/bash/lib/should_process.sh¶
#!/bin/bash

# shellcheck source=err.sh
. "${HIY_LIB:-./lib}/err.sh"

_should_process_check_mode=1

should_process_disable_check_mode() {
  _should_process_check_mode=''
}

should_process() {
  if [ "$_should_process_check_mode" ]; then
    err "[CHECK MODE] $1"
    return 1
  fi
  return 0
}

#!/bin/bash

. lib/should_process.sh

if should_process "remove dummy file"; then

fi

should_process_disable_check_mode
if should_process "remove dummy file"; then
  echo rm -v /tmp/dummy
fi

スタックトレースを出力する。¶

snippets/bash/lib/print_stacktrace.sh¶
#!/bin/bash

# shellcheck source=err.sh
. "${HIY_LIB:-./lib}/err.sh"

print_stacktrace() {
  local index=0
  local stack
  err "Traceback (most recent call first):"
  # shellcheck disable=SC2207
  while stack=($(caller "${index}")); do
    local line="${stack[0]}" sub="${stack[1]}" file="${stack[2]}"
    err "  ${file}, line ${line}, in ${sub}"
    ((index++))
  done
}
