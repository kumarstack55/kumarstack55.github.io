応用¶

ShellCheck をインストールする。¶

cd $HOME
curl -Lo - https://github.com/koalaman/shellcheck/releases/download/stable/shellcheck-stable.linux.x86_64.tar.xz | tar -Jxvf -
ln -s $HOME/shellcheck-stable/shellcheck ~/bin/shellcheck

スクリプトの絶対パスを得る。¶

#!/bin/bash

SCRIPT_PATH=$(readlink -f "${BASH_SOURCE[0]}")
echo "$SCRIPT_PATH"

標準エラーに出力する。¶

#!/bin/bash

err() {
  echo "$1" 1>&2
}

異常終了する。¶

#!/bin/bash

# shellcheck source=err.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/err.sh"

die() {
  local msg="$1"; shift
  msg="${msg:-Died}"

  local at=""
  local stack
  # shellcheck disable=SC2207
  if stack=($(caller 0)); then
    local line="${stack[0]}" file="${stack[2]}"
    at=" at ${file} line ${line}"
  fi
  err "${msg}${at}."
  exit 1
}

#!/bin/bash

# shellcheck source=lib/die.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/die.sh"

f() {
  false
}

f || die "Failed to execute f()"

$ ./demo_die.sh
Failed to execute f() at ./demo_die.sh line 10.

スタックトレースを出力する。¶



#!/bin/bash

# shellcheck source=lib/die.sh
source "$(dirname "$(readlink -f "${BASH_SOURCE[0]}")")/lib/print_stacktrace.sh"

f() {
  print_stacktrace
}

g() {
  f
}

g

コマンド実行が失敗したら異常終了する。¶

ensure() {
  if ! "$@"; then die "command failed: $*"; fi
}

#!/bin/bash

err() {
  echo "$1" 1>&2
}

die() {
  local msg="$1"; shift
  msg="${msg:-Died}"

  local at=""
  local stack
  # shellcheck disable=SC2207
  if stack=($(caller 0)); then
    local line="${stack[0]}" file="${stack[2]}"
    at=" at ${file} line ${line}"
  fi
  err "${msg}${at}."
  exit 1
}

ensure() {
  if ! "$@"; then die "command failed: $*"; fi
}

ensure false

$ ./a.sh
command failed: false at ./a.sh line 23.

コマンド実行が失敗しても無視することを明示する。¶

ignore() {
  "$@"
}

左端の空白を除去する。¶

f() {
  local msg="$1"; shift
  echo "${msg#"${msg%%[![:space:]]*}"}"
}

f "  a b c"
f "a b c"

$ f "  a b c"
a b c

$ f "a b c"
a b c
