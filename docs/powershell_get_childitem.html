停止しても再実行できるように、ファイルの一覧を出力する。¶

終了時は、ジョブキューの中身をファイルに出力する。

ロールバック可能なキューのクラス。¶

class CommitQueue {
    $List
    $Index

    CommitQueue() {
        $this.List = New-Object System.Collections.ArrayList
        $this.Index = 0
    }
    [bool]IsEmpty() {
        return ($this.List.Count - $this.Index) -eq 0
    }
    [void]Add([object]$Item) {
        $this.List.Add($Item)
    }
    [object]Get() {
        return $this.List[$this.Index++]
    }
    [void]Rollback() {
        $this.Index = 0
    }
    [void]Commit() {
        while ($this.Index -gt 0) {
            $this.List.RemoveAt(0)
            $this.Index--
        }
    }
}

class QueueFileDumper {
    [void]Dump([CommitQueue]$Queue, [string]$FullName, [bool]$Append) {
        $List = New-Object System.Collections.ArrayList
        while (-not $Queue.IsEmpty()) {
            $List.Add($Queue.Get())
        }
        $Params = @{
                'LiteralPath'=$FullName
                'Append'=$Append
                'Encoding'='UTF8'
                'NoTypeInformation'=$false
            }
        $List |
            Export-Csv @Params
    }
}

class QueueFileLoader {
    [CommitQueue]Load([string]$FullName) {
        $Queue = [CommitQueue]::new()
        Import-Csv -LiteralPath $FullName -Encoding UTF8 |
            ForEach-Object {
                $Queue.Add($_)
            }
        return $Queue
    }
}

一覧を出力する。¶

[CmdletBinding()]
param(
    [parameter(Mandatory)][string]$RootFolderPath,
    [parameter(Mandatory)][string]$QueuePath,
    [parameter(Mandatory)][string]$CsvPath
)
<#
    .SYNOPSIS
    キューファイルがない場合、RootFolderPath 以下を探索します。
    異常終了した場合、キューの内容を出力します。
    キューファイルがある場合、RootFolderPath は無視され、探索を再開します。
#>

$here = Split-Path -Parent $MyInvocation.MyCommand.Path
. "$here\CommitQueue.ps1"

class FolderQueueEntry {
    $FullName
    FolderQueueEntry([string]$FullName) {
        $this.FullName = $FullName
    }
}

class FileEntry {
    $FullName
    $Name
    $IsContainer
    $Size
    $GetChildItemFailed
    $GetAclFailed
    FileEntry([string]$FullName, [bool]$IsContainer, [long]$Size, [bool]$GetChildItemFailed, [bool]$GetAclFailed) {
        $this.FullName = $FullName
        $this.Name = Split-Path -Path $FullName -Leaf
        $this.IsContainer = $IsContainer
        $this.Size = $Size
        $this.GetChildItemFailed = $GetChildItemFailed
        $this.GetAclFailed = $GetAclFailed
    }
}

function Test-IsContainer {
    param([parameter(Mandatory)][string]$LiteralPath)
    Test-Path -Type Container -LiteralPath $LiteralPath
}

$ErrorActionPreference = 'Stop'

# 最初の１件目を登録する。CSVファイルが存在すれば消す。
if (-not (Test-Path $QueuePath)) {
    $FolderQueue1 = [CommitQueue]::new()
    $FolderQueue1.Add([FolderQueueEntry]::new($RootFolderPath))
    $Dumper = [QueueFileDumper]::new()
    $Dumper.Dump($FolderQueue1, $QueuePath, $false)

    if (Test-Path -LiteralPath $CsvPath) {
        Remove-Item $CsvPath
    }
}

# キューを読む。
$Loader = [QueueFileLoader]::new()
$FolderQueue2 = $Loader.Load($QueuePath)

try {
    while (-not $FolderQueue2.IsEmpty()) {
        $FolderPath = $FolderQueue2.Get().FullName
        Write-Warning "FolderPath: $FolderPath"

        $RecordList = New-Object System.Collections.ArrayList

        try {
            # フォルダの子アイテムを得る。
            $ItemArray = Get-ChildItem -LiteralPath $FolderPath

            # フォルダ自身をレコードにする。
            $RecordList.Add([FileEntry]::new($FolderPath, $true, 0, $false, $null)) |
                Out-Null

            # フォルダの子アイテムの内、ファイルをレコードにする。
            $ItemArray |
                Where-Object {
                    -not (Test-IsContainer -LiteralPath $_.FullName)
                } |
                ForEach-Object {
                    try {
                        Get-Acl -LiteralPath $_.FullName | Out-Null
                        $RecordList.Add([FileEntry]::new($_.FullName, $false, $_.Length, $null, $false)) |
                            Out-Null
                    } catch {
                        $RecordList.Add([FileEntry]::new($_.FullName, $false, $_.Length, $null, $true)) |
                            Out-Null
                    }
                }
        } catch [System.UnauthorizedAccessException] {
            # フォルダの子アイテムを得ることができなかった。
            $RecordList.Add([FileEntry]::new($FolderPath, $true, 0, $true, $null))
        }

        # フォルダと、フォルダの子アイテムの内、ファイルを出力する。
        $RecordList |
            Export-Csv -LiteralPath $CsvPath -Append -Encoding UTF8

        # フォルダの子アイテムの内、フォルダをキューに加える。
        if ($null -ne $ItemArray) {
            $ItemArray |
                Where-Object {
                    Test-IsContainer -LiteralPath $_.FullName
                } |
                ForEach-Object {
                    Write-Warning "Item      : $FolderPath"
                    $FolderQueue2.Add([FolderQueueEntry]::new($_.FullName))
                }
        }

        $FolderQueue2.Commit()
    }
} finally {
    $FolderQueue2.Rollback()

    # キューを保存する。
    $Dumper = [QueueFileDumper]::new()
    $Dumper.Dump($FolderQueue2, $QueuePath, $false)

    Write-Warning "キューを保存した。"
}
